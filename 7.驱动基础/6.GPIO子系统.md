# GPIO子系统 - 外设驱动开发者完整笔记

## 一、GPIO子系统概述
GPIO子系统的核心目的是为外设驱动开发者提供**硬件无关的GPIO访问接口**，屏蔽底层硬件差异，降低驱动开发难度。
核心层次（自上而下）：驱动开发者 → GPIO子系统 → GPIO控制器 → 硬件引脚
核心改进：采用**GPIO描述符（struct gpio_desc）** 替代传统GPIO编号，解决硬件相关性、编号混乱等问题，是现代驱动开发的推荐方式。

## 二、设备树配置（GPIO使用的前提）
设备树配置分为两个核心部分：GPIO控制器定义（芯片级，厂商已配置完成）和设备GPIO引用（板级/驱动级，开发者需配置）。

### 2.1 GPIO控制器定义（芯片级，无需开发者修改）
用于描述芯片上的GPIO控制器硬件信息，告知内核GPIO控制器的地址、兼容属性等。
```c
gpio1: gpio@0209c000 {
    compatible = "fsl,imx6ul-gpio", "fsl,imx35-gpio"; // 兼容属性，匹配驱动
    reg<0x0209c000 0x10000>; // 控制器寄存器地址和大小
    gpio-controller; // 标记为GPIO控制器
    #gpio-cells<2>; // 用2个cell描述一个GPIO（第一个：引脚偏移；第二个：极性/配置标志）
    interrupt-controller; // 可选，标记该GPIO控制器支持中断
    #interrupt<2>; // 中断描述所需cell数量
};
```

### 2.2 设备引用GPIO（板级/驱动级，开发者配置）
在自定义设备节点中，通过属性引用GPIO控制器的引脚，指定引脚用途、极性等。
```c
my_device {
    compatible = "mycompany,my-device"; // 设备兼容属性，匹配驱动
    
    // 1. 单个GPIO（复位引脚，高电平有效）
    reset-gpio<&gpio1 19 GPIO_ACTIVE_HIGH>;
    
    // 2. 多个GPIO（LED引脚，分别为高电平、低电平有效）
    led-gp<&gpio1 5 GPIO_ACTIVE_H<&gpio1 6 GPIO_ACTIVE_LOW>;
                
    // 3. 带中断的GPIO（上升沿触发）
    ir<&gpio1 20 GPIO_ACTIVE_RISING>;
    
    // 4. 可选GPIO（不存在时驱动不会报错）
   <&gpio1 21 GPIO_ACTIVE_HIGH>;
};
```

## 三、核心GPIO API（驱动开发核心）
GPIO API分为两大体系：现代**描述符API**（推荐，硬件无关）和传统**编号API**（兼容旧代码，逐渐废弃），以下分述其核心用法。

### 3.1 现代描述符API（推荐使用）
基于`struct gpio_desc`结构体，无需关注GPIO物理编号，直接与设备树关联，自动管理资源，降低开发成本。

#### 3.1.1 struct gpio_desc结构体（核心数据结构）
```c
struct gpio_desc {
    struct gpio_chip *chip;    // 归属：该GPIO所属的控制器
    unsigned int offset;       // 偏移：在控制器内的引脚编号（0 ~ ngpio-1）
    unsigned long flags;       // 状态标志：方向、极性、是否已申请等
    // 常见标志：GPIOF_DIR_IN（输入）、GPIOF_DIR_OUT（输出）、GPIOF_ACTIVE_LOW（低电平有效）、GPIOF_REQUESTED（已申请）
    const char *name;          // 硬件/设备树名称（如"GPIOA5"）
    const char *label;         // 驱动/功能标签（如"led-red"）
    struct device *consumer;   // 所有者：使用该GPIO的设备
    spinlock_t lock;           // 同步锁：防止并发操作
    // ... 其他私有成员
};
```

#### 3.1.2 核心API用法（含示例）
头文件`<linux/gpio/consumer.h>`，优先使用`devm_`开头的自动资源管理版本（无需手动释放）。

##### A. 获取GPIO描述符
```c
// 1. 基础获取（需手动释放）
struct gpio_desc *gpiod_get_index(struct device *dev, const char *con_id, 
                                  unsigned int index, enum gpiod_flags flags);
// 简化宏（获取单个GPIO，index=0）
#define gpiod_get(dev, con_id, flags) gpiod_get_index(dev, con_id, 0, flags)

// 2. 自动资源管理版本（推荐，probe失败自动释放）
struct gpio_desc *devm_gpiod_get_index(struct device *dev, const char *con_id, 
                                      unsigned int index, enum gpiod_flags flags);

// 3. 可选GPIO获取（不存在返回NULL，不报错）
struct gpio_desc *devm_gpiod_get_optional(struct device *dev, const char *con_id, 
                                         enum gpiod_flags flags);

// 实际示例
struct gpio_desc *reset_gpio = devm_gpiod_get(&pdev->dev, "reset", GPIOD_OUT_HIGH);
struct gpio_desc *led_gpio = devm_gpiod_get_index(&pdev->dev, "led", 1, GPIOD_OUT_LOW);
struct gpio_desc *opt_gpio = devm_gpiod_get_optional(&pdev->dev, "optional", GPIOD_OUT_LOW);
```

##### B. GPIO方向设置
```c
// 自动资源管理版本（推荐）
int devm_gpiod_direction_output(struct device *dev, struct gpio_desc *desc, int value);
// 功能：设置为输出，并指定初始值（value：0=低电平，1=高电平）

int devm_gpiod_direction_input(struct device *dev, struct gpio_desc *desc);
// 功能：设置为输入

// 手动管理版本（不推荐）
int gpiod_direction_output(struct gpio_desc *desc, int value);
int gpiod_direction_input(struct gpio_desc *desc);
```

##### C. GPIO值操作（输入/输出）
```c
// 输出操作
void gpiod_set_value(struct gpio_desc *desc, int value);        // 常规设置，考虑极性反转
void gpiod_set_raw_value(struct gpio_desc *desc, int value);    // 直接设置，忽略极性反转
void gpiod_set_value_cansleep(struct gpio_desc *desc, int value); // 可睡眠版本（适用于睡眠上下文）

// 批量输出（多个GPIO同时设置）
void gpiod_set_array_value(unsigned int array_size, struct gpio_desc **desc_array, int *values_array);

// 输入操作
int gpiod_get_value(const struct gpio_desc *desc);              // 常规读取，考虑极性反转
int gpiod_get_raw_value(const struct gpio_desc *desc);         // 直接读取，忽略极性反转

// 示例
gpiod_set_value(reset_gpio, 1);           // 输出高电平
int current_val = gpiod_get_value(irq_gpio); // 读取输入值
```

##### D. GPIO属性查询
```c
// 查询极性、方向
bool gpiod_is_active_low(const struct gpio_desc *desc);    // 是否低电平有效
bool gpiod_is_out(const struct gpio_desc *desc);           // 是否为输出方向
bool gpiod_is_in(const struct gpio_desc *desc);            // 是否为输入方向
int gpiod_get_direction(const struct gpio_desc *desc);     // 获取当前方向（返回0=输入，1=输出）

// 查询GPIO数量（如led-gpios的个数）
unsigned int gpiod_count(struct device *dev, const char *con_id);

// 示例
if (gpiod_is_out(reset_gpio)) {
    gpiod_set_value(reset_gpio, 0); // 若为输出，设置为低电平
}
unsigned int led_count = gpiod_count(&pdev->dev, "led"); // 获取led-gpios的数量
```

##### E. 中断相关操作（GPIO作为中断引脚）
```c
// 1. 从中断GPIO描述符获取IRQ号
unsigned int gpiod_to_irq(const struct gpio_desc *desc);

// 2. 申请/释放中断
int gpiod_request_direct_irq(struct gpio_desc *desc, irq_handler_t handler,
                            unsigned long flags, const char *name, void *dev);
void gpiod_free_irq(struct gpio_desc *desc, void *dev);

// 示例
int irq = gpiod_to_irq(irq_gpio); // 获取IRQ号
ret = request_irq(irq, my_interrupt_handler, IRQF_TRIGGER_RISING, "my-dev", my_dev); // 注册中断
```

### 3.2 传统编号API（兼容旧代码）
基于GPIO物理编号（unsigned int），需手动获取编号、申请资源，硬件相关性强，逐渐被描述符API替代，仅用于维护旧代码。

#### 3.2.1 核心API用法（含示例）
头文件：`#include<linux/gpio.h>`，优先使用`devm_`开头的自动资源管理版本。

##### A. 获取GPIO编号与资源申请
```c
// 1. 从设备树获取GPIO编号
int of_get_named_gpio(struct device_node *np, const char *propname, int index);

// 2. 申请GPIO（获得使用权）
int gpio_request(unsigned gpio, const char *label); // 基础申请
int gpio_request_one(unsigned gpio, unsigned long flags, const char *label); // 申请并配置方向/极性

// 3. 释放GPIO（手动释放）
void gpio_free(unsigned gpio);

// 4. 检查GPIO有效性
int gpio_is_valid(unsigned gpio);

// 5. 自动资源管理版本（推荐）
int devm_gpio_request(struct device *dev, unsigned gpio, const char *label);
int devm_gpio_request_one(struct device *dev, unsigned gpio, unsigned long flags, const char *label);

// 示例
struct device_node *np = pdev->dev.of_node;
int reset_gpio = of_get_named_gpio(np, "reset-gpio", 0); // 获取编号
if (gpio_is_valid(reset_gpio)) {
    ret = devm_gpio_request(&pdev->dev, reset_gpio, "my-reset"); // 自动申请
    if (ret) return ret;
    // 使用后无需手动释放，probe失败自动释放
}
```

##### B. GPIO方向与值操作
```c
// 方向配置
int gpio_direction_input(unsigned gpio); // 设置为输入
int gpio_direction_output(unsigned gpio, int value); // 设置为输出并指定初始值
int gpio_direction_output_raw(unsigned gpio, int value); // 直接设置，忽略极性

// 值操作
void gpio_set_value(unsigned gpio, int value); // 常规输出
void gpio_set_value_cansleep(unsigned gpio, int value); // 可睡眠版本
void gpio_set_raw_value(unsigned gpio, int value); // 直接输出，忽略极性
int gpio_get_value(unsigned gpio); // 常规读取

// 示例
gpio_direction_output(reset_gpio, 1); // 输出高电平
gpio_set_value(reset_gpio, 0); // 输出低电平
int status = gpio_get_value(in_gpio); // 读取输入值
```

## 四、核心枚举与宏定义
用于配置GPIO的方向、极性等属性，需与设备树配置、API调用配合使用。

### 4.1 gpiod_flags（描述符API专用，设置方向/初始值）
```c
enum gpiod_flags {
    GPIOD_ASIS,      // 保持设备树中原有配置，不修改方向
    GPIOD_IN,        // 设置为输入方向
    GPIOD_OUT_LOW,   // 设置为输出方向，初始低电平
    GPIOD_OUT_HIGH,  // 设置为输出方向，初始高电平
};
```

### 4.2 GPIO极性/配置宏（设备树专用）
```c
// 设备树中描述GPIO属性的宏，用于第二个cell
#define GPIO_ACTIVE_HIGH 0   // 高电平有效（默认）
#define GPIO_ACTIVE_LOW  1   // 低电平有效
#define GPIO_OPEN_DRAIN  (1 << 1)    // 开漏输出
#define GPIO_OPEN_SOURCE (1 << 2)    // 开源输出
#define GPIO_PULL_UP     (1 << 3)    // 上拉配置
#define GPIO_PULL_DOWN   (1 << 4)    // 下拉配置
```

## 五、完整驱动示例（两种API对比）


### 5.1 现代描述符API示例（推荐）
```c
#include<linux/platform_device.h><linux/gpio/consumer.h>
#include<linux/interrupt.h>

struct my_device_data {
    struct gpio_desc *reset_gpio;  // 复位GPIO
    struct gpio_desc *enable_gpio; // 使能GPIO
    struct gpio_desc *status_gpio; // 状态GPIO（输入）
    struct gpio_desc *irq_gpio;    // 中断GPIO
    int irq_num;                   // 中断号
};

// 中断处理函数
static irqreturn_t my_irq_handler(int irq, void *dev_id)
{
    struct my_device_data *data = dev_id;
    int status = gpiod_get_value(data->status_gpio);
    printk(KERN_INFO "IRQ triggered, device status: %d\n", status);
    return IRQ_HANDLED;
}

// probe函数（驱动初始化）
static int my_device_probe(struct platform_device *pdev)
{
    struct my_device_data *data;
    int ret;
    
    // 分配设备私有数据
    data = devm_kzalloc(&pdev->dev, sizeof(*data), GFP_KERNEL);
    if (!data) return -ENOMEM;
    platform_set_drvdata(pdev, data);//将驱动的私有数据结构体指针关联到platform_device
    
    // 1. 获取所有GPIO描述符
    data->reset_gpio = devm_gpiod_get(&pdev->dev, "reset", GPIOD_OUT_HIGH);
    if (IS_ERR(data->reset_gpio)) return PTR_ERR(data->reset_gpio);
    
    data->enable_gpio = devm_gpiod_get(&pdev->dev, "enable", GPIOD_OUT_LOW);
    if (IS_ERR(data->enable_gpio)) return PTR_ERR(data->enable_gpio);
    
    data->status_gpio = devm_gpiod_get(&pdev->dev, "status", GPIOD_IN);
    if (IS_ERR(data->status_gpio)) return PTR_ERR(data->status_gpio);
    
    data->irq_gpio = devm_gpiod_get(&pdev->dev, "irq", GPIOD_IN);
    if (IS_ERR(data->irq_gpio)) return PTR_ERR(data->irq_gpio);
    
    // 2. 初始化设备（复位+使能）
    gpiod_set_value(data->reset_gpio, 0);  // 拉低复位
    msleep(10);                            // 延时复位
    gpiod_set_value(data->reset_gpio, 1);  // 释放复位
    gpiod_set_value(data->enable_gpio, 1); // 使能设备
    
    // 3. 注册中断
    data->irq_num = gpiod_to_irq(data->irq_gpio);
    ret = request_irq(data->irq_num, my_irq_handler, 
                      IRQF_TRIGGER_RISING, "my-device", data);
    if (ret) {
        dev_err(&pdev->dev, "Failed to request IRQ\n");
        return ret;
    }
    
    dev_info(&pdev->dev, "Device probe success\n");
    return 0;
}

// remove函数（驱动卸载）
static int my_device_remove(struct platform_device *pdev)
{
    struct my_device_data *data = platform_get_drvdata(pdev);
    free_irq(data->irq_num, data); // 释放中断
    return 0;
}

// 设备树匹配表
static const struct of_device_id my_device_of_match[] = {
    { .compatible = "mycompany,my-device", },
    { }
};
MODULE_DEVICE_TABLE(of, my_device_of_match);

// platform驱动结构体
static struct platform_driver my_device_driver = {
    .probe = my_device_probe,
    .remove = my_device_remove,
    .driver = {
        .name = "my-device",
        .of_match_table = my_device_of_match,
    },
};

module_platform_driver(my_device_driver);
MODULE_LICENSE("GPL");
MODULE_DESCRIPTION("GPIO Subsystem Modern API Example");
```

### 5.2 传统编号API示例（兼容旧代码）
```c
#include <linux/init.h>
#include <linux/module.h>
#include <linux/platform_device.h>
#include<linux/gpio.h<linux/of_gpio.h>

static int legacy_driver_probe(struct platform_device *pdev)
{
    struct device_node *np = pdev->dev.of_node;
    int reset_gpio, led_gpio;
    int ret;
    
    // 1. 从设备树获取GPIO编号
    reset_gpio = of_get_named_gpio(np, "reset-gpio", 0);
    led_gpio = of_get_named_gpio(np, "led-gpio", 0);
    
    // 2. 检查GPIO有效性
    if (!gpio_is_valid(reset_gpio)) {
        dev_err(&pdev->dev, "Invalid reset GPIO\n");
        return -EINVAL;
    }
    if (!gpio_is_valid(led_gpio)) {
        dev_err(&pdev->dev, "Invalid led GPIO\n");
        return -EINVAL;
    }
    
    // 3. 申请GPIO（自动资源管理）
    ret = devm_gpio_request(&pdev->dev, reset_gpio, "my-reset");
    if (ret) {
        dev_err(&pdev->dev, "Failed to request reset GPIO\n");
        return ret;
    }
    ret = devm_gpio_request(&pdev->dev, led_gpio, "my-led");
    if (ret) {
        dev_err(&pdev->dev, "Failed to request led GPIO\n");
        return ret;
    }
    
    // 4. 配置GPIO方向并使用
    gpio_direction_output(reset_gpio, 1); // 复位引脚输出高电平（释放复位）
    gpio_direction_output(led_gpio, 0);   // LED引脚输出低电平（熄灭）
    msleep(1000);
    gpio_set_value(led_gpio, 1);          // LED点亮
    
    dev_info(&pdev->dev, "Legacy driver probe success\n");
    return 0;
}

static int legacy_driver_remove(struct platform_device *pdev)
{
    // 无需手动释放GPIO，devm自动管理
    return 0;
}

static const struct of_device_id legacy_of_match[] = {
    { .compatible = "mycompany,legacy-device", },
    { }
};
MODULE_DEVICE_TABLE(of, legacy_of_match);

static struct platform_driver legacy_device_driver = {
    .probe = legacy_driver_probe,
    .remove = legacy_driver_remove,
    .driver = {
        .name = "legacy-device",
        .of_match_table = legacy_of_match,
    },
};

module_platform_driver(legacy_device_driver);
MODULE_LICENSE("GPL");
MODULE_DESCRIPTION("GPIO Subsystem Legacy API Example");
```

## 六、API对比与选择指南
明确两种API的差异，根据开发场景选择合适的方式，优先使用现代描述符API。

| 特性 | 传统编号API | 现代描述符API |
|------|-------------|---------------|
| 头文件 |<linux/gpio.h><linux/gpio/consumer.h> |
| 核心数据类型 | unsigned int（GPIO编号） | struct gpio_desc *（GPIO描述符） |
| 硬件相关性 | 强（依赖物理编号） | 弱（硬件无关，与设备树关联） |
| 设备树集成 | 需额外调用of_get_named_gpio获取编号 | 直接通过con_id关联设备树属性，无需额外操作 |
| 推荐程度 | 逐渐废弃，仅用于维护旧代码 | 强烈推荐，新驱动优先使用 |
| 代码存在度 | 大量现有旧代码中存在 | 新开发代码的标准用法 |
