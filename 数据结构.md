### 搜索二叉树
定义：左子树所有值 < 根 < 右子树所有值
**前序**
- 根 → 左 → 右：可重构还原一颗树
```cpp
void dfs(TreeNode* root){
    if(!root) return;
    处理root;    //根
    dfs(root->left); //左
    dfs(root->right);//右
}
```
或者
```cpp
//根入栈
//出栈就处理
//因为栈后进先出：先压右，再压左
//这样弹出才是 根→左→右
vector<int> preorder(TreeNode* root) {
    vector<int> res;
    stack<TreeNode*> st;
    if(root) st.push(root);

    while(!st.empty()){
        TreeNode* cur = st.top();
        st.pop();

        res.push_back(cur->val); // 出栈立刻处理

        if(cur->right) st.push(cur->right); // 先压右
        if(cur->left)  st.push(cur->left);  // 后压左
    }
    return res;
}
```

**中序**

- 左 → 根 → 右：得到升序结果，反向中序得到降序结果
```cpp
void dfs(TreeNode* root){
    if(!root) return;
    dfs(root->left); //左
    处理root;    //根
    dfs(root->right);//右
}
```
或者
```cpp
//一路向左走到底，全部压栈
//走不动了（cur=null），弹出栈顶 = 处理根
//转向当前节点右子树，重复循环
vector<int> inorder(TreeNode* root) {
    vector<int> res;
    stack<TreeNode*> st;
    TreeNode* cur = root;

    while(cur || !st.empty()){
        // 1. 一路向左，全压栈
        while(cur){
            st.push(cur);
            cur = cur->left;
        }

        // 2. 左边走完，弹出，处理节点
        cur = st.top();
        st.pop();
        res.push_back(cur->val);

        // 3. 去右子树
        cur = cur->right;
    }
    return res;
}
```
**后序**
- 左 → 右 → 根：最后一个数字是整棵树的根，可用于拆分大于根的子树和小于根的子树
```cpp
void dfs(TreeNode* root){
    if(!root) return;
    dfs(root->left); //左
    dfs(root->right);//右
    处理root;    //根
}
```
或者
```cpp
//前序是：根 左 右
//我们改成：根 右 左
//最后把数组反转 = 左 右 根 = 后序
vector<int> postorder(TreeNode* root) {
    vector<int> res;
    stack<TreeNode*> st;
    if(root) st.push(root);

    while(!st.empty()){
        TreeNode* cur = st.top();
        st.pop();
        res.push_back(cur->val);

        if(cur->left)  st.push(cur->left);  // 先压左
        if(cur->right) st.push(cur->right); // 后压右
    }
    reverse(res.begin(), res.end());
    return res;
}
```
**层序**
- 从上到下、一层一层遍历
```cpp
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        dfs(root, 0, res);
        return res;
    }

    // level：当前节点层数，从0开始
    void dfs(TreeNode* root, int level, vector<vector<int>>& res) {
        if(root == nullptr) return;

        // 如果这一层还没创建数组，新开一层
        if(res.size() == level){
            res.push_back(vector<int>());
        }

        // 把当前节点加入对应层数
        res[level].push_back(root->val);

        // 递归下一层：左→右
        dfs(root->left, level + 1, res);
        dfs(root->right, level + 1, res);
    }
};
```
或者
```cpp
vector<vector<int>> levelOrder(TreeNode* root) {
    vector<vector<int>> res;
    queue<TreeNode*> q;

    if(root) q.push(root);

    while(!q.empty()){
        int size = q.size(); // 当前一层有几个节点
        vector<int> level;

        for(int i=0;i<size;i++){
            TreeNode* cur = q.front();
            q.pop();

            level.push_back(cur->val); // 访问当前节点

            if(cur->left)  q.push(cur->left);  // 先左
            if(cur->right) q.push(cur->right); // 后右
        }
        res.push_back(level);
    }
    return res;
}
```
